// Cloudflare Worker — Proxy Infosimples DETRAN-MG // Cole este código no editor do Cloudflare Workers export default { async fetch(request, env) { // Permite CORS para qualquer origem (necessário para o GitHub Pages chamar) const corsHeaders = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Methods': 'GET, POST, OPTIONS', 'Access-Control-Allow-Headers': 'Content-Type', }; // Responde preflight OPTIONS if (request.method === 'OPTIONS') { return new Response(null, { headers: corsHeaders }); } // Só aceita POST if (request.method !== 'POST') { return new Response(JSON.stringify({ erro: 'Use POST' }), { status: 405, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }); } let body; try { body = await request.json(); } catch (e) { return new Response(JSON.stringify({ erro: 'JSON inválido' }), { status: 400, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }); } const { token, placa, chassi, renavam, login_cpf, login_senha } = body; if (!token || !placa || !chassi || !login_cpf || !login_senha) { return new Response(JSON.stringify({ erro: 'Parâmetros obrigatórios: token, placa, chassi, login_cpf, login_senha' }), { status: 400, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }); } // Monta URL da Infosimples const params = new URLSearchParams({ token, placa: placa.replace(/[^A-Z0-9]/gi, '').toUpperCase(), chassi: chassi.trim().toUpperCase(), renavam: renavam ? renavam.trim() : '', login_cpf: login_cpf.replace(/\D/g, ''), login_senha, }); const url = `https://api.infosimples.com/api/v2/consultas/detran/mg/multas-descritivos?${params}`; try { const resp = await fetch(url, { method: 'GET', headers: { 'User-Agent': 'CloudflareWorker/1.0' } }); const data = await resp.json(); return new Response(JSON.stringify(data), { status: 200, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }); } catch (e) { return new Response(JSON.stringify({ erro: 'Falha ao chamar Infosimples: ' + e.message }), { status: 502, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }); } } };
